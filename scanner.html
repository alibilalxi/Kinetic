function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var fullEmail = data.emailUser + "@" + data.emailDomain; var dataRange = sheet.getDataRange().getValues(); for (var i = 1; i < dataRange.length; i++) { if (dataRange[i][1] == data.fullName || dataRange[i][2] == data.nationalId || dataRange[i][4] == fullEmail) { return ContentService.createTextOutput(JSON.stringify({ "status": "duplicate", "barcode": dataRange[i][7], "choiceName": dataRange[i][5], "price": dataRange[i][6] })).setMimeType(ContentService.MimeType.JSON); } } var barcodeValue = data.barcode; sheet.appendRow([ new Date(), data.fullName, data.nationalId, data.status, fullEmail, data.itemChoice, data.price, barcodeValue, "Pending" ]); return ContentService.createTextOutput(JSON.stringify({ "status": "success", "barcode": barcodeValue, "choiceName": data.itemChoice, "price": data.price })).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if(e.parameter.action == "fetchAll") { var values = sheet.getDataRange().getValues(); var data = []; for (var i = 1; i < values.length; i++) { data.push({ name: values[i][1], choice: values[i][5], price: values[i][6], barcode: values[i][7], status: values[i][8] }); } return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } if(e.parameter.action == "scan") { var dataRange = sheet.getDataRange().getValues(); var codeToFind = e.parameter.barcode; for (var i = 1; i < dataRange.length; i++) { if (dataRange[i][7] == codeToFind) { var rowIndex = i + 1; // رجعنا التاريخ والوقت كاملين زي الأول بالظبط var now = new Date(); var days = ["الأحد", "الإثنين", "الثلاثاء", "الأربعاء", "الخميس", "الجمعة", "السبت"]; var months = ["يناير", "فبراير", "مارس", "أبريل", "مايو", "يونيو", "يوليو", "أغسطس", "سبتمبر", "أكتوبر", "نوفمبر", "ديسمبر"]; var formattedDate = days[now.getDay()] + " " + now.getDate() + " " + months[now.getMonth()]; var hours = now.getHours(); var minutes = now.getMinutes(); var ampm = hours >= 12 ? 'م' : 'ص'; hours = hours % 12; hours = hours ? hours : 12; minutes = minutes < 10 ? '0' + minutes : minutes; var formattedTime = hours + ':' + minutes + ' ' + ampm; sheet.getRange(rowIndex, 9).setValue("Paid"); sheet.getRange(rowIndex, 10).setValue(formattedDate); sheet.getRange(rowIndex, 11).setValue(formattedTime); // عمود الوقت sheet.getRange(rowIndex, 9, 1, 3).setBackground("#d4edda"); return ContentService.createTextOutput(JSON.stringify({ "status": "success" })).setMimeType(ContentService.MimeType.JSON); } } } }
